        function generateWaveform(samples, duration) {
            const dt = duration / samples;
            const t = Array.from({ length: samples }, (_, i) => i * dt);
            return t.map(ti => {
                const noise = (Math.random() - 0.5) * 0.1;
                const p1 = Math.exp(-0.6 * ti) * Math.sin(2 * Math.PI * 2.0 * ti);
                const p2 = 0.6 * Math.exp(-0.8 * (ti - 2)) * Math.sin(2 * Math.PI * 5.0 * (ti - 2)) * (ti > 2 ? 1 : 0);
                return p1 + p2 + noise;
            });
        }

        function applyFilter(signal, ampFactor, damping) {
            const dt = 8 / signal.length;
            const t = Array.from({ length: signal.length }, (_, i) => i * dt);
            return signal.map((val, i) => val * ampFactor * Math.exp(-damping * t[i]));
        }

        function renderPath(id, signal, width, height) {
            const y = (s, h) => {
                const max = Math.max(...s.map(Math.abs)) * 1.2 || 1;
                return s.map(v => h * 0.5 * (1 - v / max));
            };
            const yPoints = y(signal, height);
            const xStep = width / (signal.length - 1);
            let d = `M 0 ${yPoints[0]}`;
            for (let i = 1; i < signal.length; i++) d += ` L ${i * xStep} ${yPoints[i]}`;
            const path = document.getElementById(id);
            if (path) path.setAttribute("d", d);
        }

            const soilSignal1 = applyFilter(bedrockSignal, 2.0, 0.07);
            const surfaceSignal = applyFilter(bedrockSignal, 2.5, 0.05);
            
            const signals = {
                'surface-wave-path': surfaceSignal,
                'soil-wave-path-1': soilSignal1,
                'soil-wave-path-2': soilSignal2,
                'bedrock-wave-path': bedrockSignal
            };

                        const signal = signals[pathEl.id];
                        if (signal) {
                            renderPath(pathEl.id, signal, width, height);
                        }
                    }
                }
            });

                    }
                });
            }
        });
